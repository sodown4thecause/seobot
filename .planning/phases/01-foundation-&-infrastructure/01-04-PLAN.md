<objective>
Implement three-layer caching infrastructure for dashboard data with time-to-live (TTL) by data type and stale-while-revalidate pattern.

Purpose: Dashboards must load within 3 seconds. Multi-layer caching achieves this by storing frequently-accessed data at the edge (Next.js cache), in the browser (TanStack Query), and in Redis. Stale-while-revalidate ensures users always see data while background refresh happens.
Output: Complete caching layer with Redis client, TanStack Query setup, and React hooks for dashboard data.
</objective>

```typescript
export const dashboardQueryClient = new QueryClient({
  defaultOptions: {
    queries: {
      // Stale time by data type (matches Redis TTL)
      staleTime: 1000 * 60 * 5, // 5 minutes default
      gcTime: 1000 * 60 * 60, // 1 hour
      retry: 2,
      refetchOnWindowFocus: false,
    },
  },
})

// Helper to get staleTime for specific data types
export function getStaleTimeForDataType(dataType: string): number {
  switch (dataType) {
    case 'rankings': return 1000 * 60 * 60 * 6 // 6 hours
    case 'backlinks': return 1000 * 60 * 60 * 48 // 48 hours
    case 'audit': return 1000 * 60 * 60 * 24 // 24 hours
    case 'competitor': return 1000 * 60 * 60 * 12 // 12 hours
    default: return 1000 * 60 * 5 // 5 minutes
  }
}
```
